64 bit left-logical shift
    LDRD R0,R1,x
    LSL R1,R1,#1
    ORR R1,R1,R0, LSL#31
    LSL R0,R0,#1
    STRD R0,R1,x
    
64-bit logical shift right
    LDRD R0,R1,x
    LSR R0,R0,#1
    ORR R0,R0,R1,LSL#31
    LSR R`1,R1,#1
    STRD R0,R1,x
    
    ;also use RRX
    LDRD R0,R1,x
    LSRS R1,R1,#1
    RRX R0,R0
    STRD R0,R1,x
    
64-bit Left Rotate
    LDRD R0,R1,x
    LSLS R0,R0,#1
    ORR R0,R0,R1,LSR#31
    ADC R1,R1,R1
    STRD R0,R1,x
